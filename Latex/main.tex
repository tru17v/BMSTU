\section*{Задание 1}
    \subsection*{Условие}
        Вычислить неопределенный интеграл от функции $y=\frac{1}{e^{x}+1}$, выполнив замену переменной $x=-\ln{t}$.
    
    \subsection*{Решение}
        \begin{displaymath}
            \begin{split}
                &\int{\frac{1}{e^{x}+1}\,dx}=
                    \begin{array}{|l|}
                     x=-\ln{t}\\
                     dx=-\,dt/t\\
                     t=e^{-x} 
                    \end{array}=
                -\int{\frac{dt}{t(1/t+1)}}=
                -\int{\frac{dt}{1+t}}=
                -\int{\frac{d(t+1)}{t+1}}=\\
                &=-\ln{(t+1)}+C=
                -\ln{|e^{-x}+1|}+C.
            \end{split}
        \end{displaymath}\\
